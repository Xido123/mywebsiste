vue监听属性

2019.11.27

1.实例

段落

<div id="app">
            <p style = "font-size: 25px;">计数器：{{ counter }}</p>
            <button @click = "counter++" style = "font-size: 25px;" >点我</button>
        </div>
        <script>
        var vm= new Vue({
            el:'#app',
            data:{
                counter:1
            }
        });
        vm.$watch('counter',function(nval,oval){
            alert('计数器值的变化：' + oval + '变为' + nval + '!')
        })

        </script>
